class PRNG:
    def __init__(self, seed = 327680):
        self.seed = seed
        self.a = 4291010243
        self.c = 179203

    def random(self):
        self.seed = (self.a - self.seed * self.c) % 16777216
        rnd = self.seed / 16777216
        return rnd

rand = PRNG(0)
